1 /**
  2  * @module TangoAttribute
  3  */
  4 (function () {
  5     /**
  6      * Model tango_attribute
  7      *
  8      * @type {TangoAttribute}
  9      */
 10     TangoAttribute = MVC.Model.extend('tango_attribute',
 11         /** @Static */
 12         {
 13 
 14             attributes: {
 15                 id: 'string',//host_id/device_id/name
 16                 name: 'string',
 17                 device_id: 'string',
 18                 display_name: 'string'
 19                 //TODO value
 20             },
 21             default_attributes: {},
 22             /**
 23              *
 24              * @param id
 25              * @return {{tango_host: string, tango_port: number, device: string, name: string}}
 26              */
 27             parseId:function(id){
 28                 var parts = id.split('/');
 29 
 30                 return {
 31                     id: id,
 32                     host:parts[0],
 33                     device:[parts[1],parts[2], parts[3]].join('/'),
 34                     name:parts[4]
 35                 }
 36             }
 37         },
 38         /** @Prototype */
 39         {
 40             /**
 41              *
 42              * @param attrs
 43              * @constructor
 44              */
 45             init:function(attrs){
 46                 attrs.display_name = attrs.display_name || attrs.name;
 47                 this._super(attrs);
 48             },
 49             /**
 50              *
 51              * @return {boolean}
 52              */
 53             isScalar:function(){
 54                 return this.info.data_format === 'SCALAR';
 55             },
 56             _handle_resp : function (resp) {
 57                 var result = resp[0];
 58                 this.update_attributes(result);
 59                 if (!this.valid())
 60                     throw this;
 61                 else return this;
 62             },
 63             /**
 64              * @returns {webix.promise}
 65              */
 66             read: function () {
 67                 var device = PlatformContext.devices.getItem(this.device_id);
 68 
 69 
 70                 return device.fetchAttrValues([this.name]).then(this._handle_resp.bind(this));
 71             },
 72             /**
 73              *
 74              * @returns {webix.promise}
 75              */
 76             fetchHistory:function(){
 77                 var device = PlatformContext.devices.getItem(this.device_id);
 78                 return device.toTangoRestApiRequest().attributes(this.name).get('/history')
 79                     .then(function(resp){
 80                         this.update_attributes({
 81                             history: resp,
 82                             timestamp: +new Date()
 83                         });
 84                         return this;
 85                     }.bind(this));
 86             },
 87             /**
 88              *
 89              * @param value
 90              * @returns {webix.promise}
 91              */
 92             write: function (value) {
 93                 var device = PlatformContext.devices.getItem(this.device_id);
 94 
 95                 var values = {};
 96                 values[this.name] = value;
 97                 return device.putAttrValues(values).then(this._handle_resp.bind(this));
 98             },
 99             /**
100              *
101              * @returns {*|webix.promise}
102              */
103             //TODO extract AttributeInfo (aka MVC.Model.JSON) and move this method there
104             putInfo: function () {
105                 var device = PlatformContext.devices.getItem(this.device_id);
106 
107                 return device.toTangoRestApiRequest().attributes(this.name).put('/info?async=true', this.info);
108             },
109             /**
110              *
111              * @returns {'STATE'|'STATUS'|'SCALAR'|'SPECTRUM'|'IMAGE'}
112              */
113             getDataFormat:function(){
114                 if(this.name === 'State') return 'STATE';
115                 if(this.name === 'Status') return 'STATUS';
116                 return this.info.data_format;
117             }
118         }
119     );
120 })();
