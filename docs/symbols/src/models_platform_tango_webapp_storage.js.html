1 /**
  2  * Model tango_webapp_storage
  3  *
  4  * @type {TangoWebappStorage}
  5  */
  6 TangoWebappStorage = MVC.Class.extend(
  7     /* @prototype */
  8     {
  9         /**
 10          *
 11          * @param {Object} klass
 12          */
 13         init: function (klass) {
 14             this.key = klass.className;
 15             this.storing_class = klass;
 16             if(localStorage.getItem(this.key) == null) {
 17                 localStorage.setItem(this.key, "{}");
 18             }
 19         },
 20         /**
 21          *
 22          * @param {Object} id
 23          */
 24         find_one: function (id) {
 25             return id ? JSON.parse(localStorage.getItem(this.key))[id] : null;
 26         },
 27         /**
 28          *
 29          * @param {Object} obj
 30          */
 31         create: function (obj) {
 32             var id = obj[obj.Class.id];
 33             this.update(id, obj.attributes());
 34         },
 35         update: function (id, attrs) {
 36             var data = JSON.parse(localStorage.getItem(this.key));
 37             data[id] = attrs;
 38             localStorage.setItem(this.key, JSON.stringify(data));
 39         },
 40         /**
 41          *
 42          * @param {Object} id
 43          */
 44         destroy: function (id) {
 45             var data = JSON.parse(localStorage.getItem(this.key));
 46             delete data[id];
 47             localStorage.setItem(this.key, JSON.stringify(data));
 48         },
 49         /**
 50          * Finds instances using a test function.  If no test function is provided returns all instances.
 51          * @param {Function} f
 52          * @return {Array}
 53          */
 54         find: function (f) {
 55             var instances = [];
 56             var data = JSON.parse(localStorage.getItem(this.key));
 57             for (var id in data) {
 58                 var inst = data[id];
 59                 if (!f || f(inst))
 60                     instances.push(inst);
 61             }
 62             return this.storing_class.create_many_as_existing(instances);
 63         },
 64         /**
 65          * Clears instances
 66          */
 67         clear: function () {
 68             localStorage.setItem(this.key, "{}");
 69         },
 70         /**
 71          * Returns if there is no instances
 72          * @return {Boolean}
 73          */
 74         is_empty: function () {
 75             return !this.find().length;
 76         }
 77     });
