227     var attrs_monitor_view = webix.protoUI(
228         /**
229          * @lends AttrsMonitorView.prototype
230          */
231         {
232         name: 'attrs_monitor',
233         _monitored: null,
234         _plotted: null,
235         _devices: null,
236         _plotIndex: 0, //TODO is this shared?
237         _add_attr: function (attr) {
238             if (attr.info.data_format === 'SPECTRUM') {
239                 this.$$('attributes').addView({
240                     close:true,
241                     header: attr.info.label,
242                     body: TangoWebapp.ui.newSpectrumView(attr)
243                 });
244             } else if (attr.info.data_format === 'IMAGE') {
245                 this.$$('attributes').addView({
246                     close:true,
247                     header: attr.info.label,
248                     body: TangoWebapp.ui.newImageView(attr)
249                 });
250             }
251         },
252             /**
253              * 
254              * @param id
255              */
256         removeItem:function(id){
257             this._monitored.remove(id);
258             if(this.$$('scalars').exists(id))
259                 this.$$('scalars').remove(id);
260         },
261         /**
262          *
263          * @param {} item
264          */
265         startPlot:function(item){
266             var $$plot = this.$$('plot');
267             var $$scalars = this.$$('scalars');
268 
269             $$scalars.updateItem(item.id, {
270                 plotted: true,
271                 plotIndex: this._plotIndex++
272             });
273 
274             var device = PlatformContext.devices.getItem(item.device_id);
275             $$plot.addTrace(device.display_name + "/" + item.label, [item.timestamp], [item.value], item.plotIndex);
276         },
277         /**
278          *
279          * @param item
280          */
281         stopPlot:function(item){
282             var $$plot = this.$$('plot');
283             var $$scalars = this.$$('scalars');
284 
285             var index = item.plotIndex;
286 
287             $$plot.deleteTrace(index);
288 
289             $$scalars.updateItem(item.id, {
290                 plotted: false,
291                 plotIndex: undefined
292             });
293 
294             //TODO replace plotted with array?
295             TangoWebappHelpers.iterate(this._plotted, function(plotted){
296                 if(plotted.plotIndex >= index) plotted.plotIndex--;
297             });
298 
299             this._plotIndex--;
300         },
301         _update_plot:function(){
302             if (this._plotted.count() === 0) return;
303 
304             var plotted =[];
305             TangoWebappHelpers.iterate(this._plotted, function(scalar){
306                 plotted.push(scalar);
307             });
308             this.$$('plot').updateTraces(
309                 plotted.map(function (el) {
310                     return el.plotIndex;
311                 }),
312                 plotted.map(function (el) {
313                     return el.timestamp;
314                 }),
315                 plotted.map(function (el) {
316                     return el.value;
317                 })
318             );
319         },
320         /**
321          *
322          */
323         update: function (attrs) {
324             attrs.forEach(function (attr) {
325                 if (attr.info.data_format === 'SCALAR' && this.$$('scalars').exists(attr.id)) {
326                     this.$$('scalars').updateItem(attr.id, attr);
327                 } else if(this.$$(attr.id) !== undefined) {
328                     this.$$(attr.id).update(attr);
329                 }
330             }, this);
331 
332             this._update_plot();
333         },
334         _update_start_stop_icon:function(icon){
335             this.$$('startStop').define("icon", icon);
336             this.$$('startStop').refresh();
337         },
338         before_start:function(){
339             this._update_start_stop_icon("pause");
340             webix.html.addCss( this.$$('status').getNode(), "fa-spin");
341         },
342         after_stop:function(){
343             this._update_start_stop_icon("play");
344             webix.html.removeCss( this.$$('status').getNode(), "fa-spin");
345         },
346         /**
347          *
348          * @return {[]}
349          * @private
350          */
351         _get_attrs_to_update:function(){
352             var result = [];
353             TangoWebappHelpers.iterate(this._plotted, function (plotted) {
354                 result.push(
355                     this._monitored.getItem(plotted.id));
356             }.bind(this));
357 
358             if (this.$$('attributes').getTabbar().getValue() === 'scalars') {
359                 TangoWebappHelpers.iterate(this.$$('scalars').data, function (scalar) {
360                     if (!scalar.plotted) result.push(this._monitored.getItem(scalar.id));
361                 }.bind(this));
362             } else {
363                 result.push(
364                     this._monitored.getItem(this.$$('attributes').getTabbar().getValue()));
365             }
366             return result;
367         },
368         run: function () {
369             var attrs_to_update = this._get_attrs_to_update();
370 
371             for (var device_id in this._devices) {
372                 var device = this._devices[device_id];
373 
374                 var filtered_attrs_to_update = attrs_to_update
375                     .filter(function (attr_to_update) {
376                         return attr_to_update.device_id === device.id;
377                     });
378                 
379                 if(filtered_attrs_to_update.length !== 0)
380                     device.fetchAttrValues(filtered_attrs_to_update
381                         .map(function (attr_to_update) {
382                             return attr_to_update.name;
383                         })).then(function (filtered_attrs_to_update, resp) {
384                             this.update(filtered_attrs_to_update.map(function (filtered_attr_to_update, ndx) {
385                                 return MVC.Object.extend(filtered_attr_to_update, resp[ndx]);
386                             }));
387                     }.bind(this, filtered_attrs_to_update.slice()))
388                         .fail(function(err){
389                             TangoWebappHelpers.error(err);
390                         });
391             }
392         },
393         /**
394          *
395          * @param {[]}  scalars
396          */
397         parseScalars:function(scalars){
398             this._monitored.parse(scalars);
399             this.$$('scalars').parse(scalars.map(function(scalar){
400                 return {
401                     id: scalar.id,
402                     device_id: scalar.device_id,
403                     label: scalar.info.label,
404                     unit: scalar.info.unit,
405                     description: scalar.info.description
406                 }
407             }));
408 
409             this.$$('scalars').hideOverlay();
410         },
411         /**
412          * @param {TangoDevice} device
413          */
414         addDevice:function(device){
415             this._devices[device.id] = device;
416         },
417         /**
418          *
419          * @param {TangoAttribute} attr
420          * @return boolean
421          */
422         addAttribute: function (attr) {
423             if (this._monitored.getItem(attr.id) !== undefined) return false;
424 
425             attr = attr.attributes();
426             this._monitored.add(attr);
427 
428             if (attr.info.data_format !== 'SCALAR') {
429                 this._add_attr(attr);
430             } else {
431                 this.$$('scalars').addAttribute(attr);
432             }
433 
434             this.addDevice(PlatformContext.devices.getItem(attr.device_id)); //sync filtered?
435 
436             return true;
437         },
438         /**
439          *
440          * @param {TangoAttribute} attr
441          */
442         removeAttribute:function(attr){
443             this.removeItem(attr.id);
444         },

463              * @constructs
464              * @param config
465              */
466         $init: function (config) {
467             webix.extend(config, this._ui());
468 
469             this._devices = Object.create(null);
470             this._monitored = new webix.DataCollection();
471             this._plotted = new webix.DataCollection();
472 
473             this.$ready.push(function () {
474                 this._plotted.sync(this.$$('scalars').data, function () {
475                     this.filter(function (obj) {
476                         return obj.plotted;
477                     })
478                 });
479 
480                 this.$$('attributes').getTabbar().attachEvent("onBeforeTabClose",function(id){
481                     this.removeItem(id);
482                 }.bind(this))
483             }.bind(this));
484         },
485         defaults:{
486             on:{
487                 "tango_webapp.item_selected subscribe":function(event){
488                     if(event.data.kind !== 'attrs') return;
489                     var self = event.controller;
490                     if(self.$$('scalars').exists(event.data.id) && self.$$('scalars').getSelectedId() !== event.data.id)
491                         self.$$('scalars').select(event.data.id);
492                 }
493             }
494         }

506     var AttrsMonitorState = TangoWebappPlatform.WidgetState.extend(
507         {
508             init:function(attrs){
509                 attrs.data = attrs.data || Object.create(null);
510                 this._super(attrs);
511             },
512             /**
513              * @return {Array}
514              */
515             asIdArray:function(){
516                 var result = [];
517                 for(var id in this.data)
518                     result.push(id);
519                 return result;
520             },
521             updateState:function(id, plotted){
522                 var item = Object.create(null);
523                 item[id] = plotted;
524                 this._super(item);
525             }
526         }

529     /**
530      * @name StatefulAttrsMonitor
531      * @type {webix.protoUI}
532      * @extends AttrsMonitorView
533      * @class
534      */
535     var stateful_attrs_monitor = webix.protoUI(
536         /** @lends StatefulAttrsMonitor.prototype*/
537         {
538         name:'stateful_attrs_monitor',
539         /**
540          *
541          * @param {AttrsMonitorState} state atributes' names
542          */
543         restoreState:function(state){
544             var parsedIds = state.asIdArray().map(function(id){
545                 return TangoAttribute.parseId(id);
546             });
547 
548             //unique host -> unique devices
549             var unique_hosts = Object.create(null);
550             parsedIds.forEach(function(id){
551                 if(unique_hosts[id.host] === undefined)
552                     unique_hosts[id.host] = Object.create(null);
553 
554                 unique_hosts[id.host][id.device] = null;
555             });
556 
557 
558             var self = this;
559             for(var host in unique_hosts)
560                 for (var device in unique_hosts[host])
561                     (function(host, device) {
562                         PlatformContext.rest.fetchHost(host)
563                             .then(function (host) {
564                                 return host.fetchDevice(device);
565                             })
566                             .then(function (device) {
567                                 return device.fetchAttrs();
568                             })
569                             .then(function (attrs) {
570                                 var ids = state.asIdArray();
571                                 attrs.filter(function (attr) {
572                                     return ids.indexOf(attr.id) > -1;
573                                 }).forEach(function (attr) {
574                                     webix.ui.attrs_monitor.prototype.addAttribute.apply(self, arguments);
575                                     if(state.data[attr.id])
576                                         webix.ui.attrs_monitor.prototype.startPlot.call(self, self.$$('scalars').getItem(attr.id));
577                                 });
578                             })
579                             .fail(function(err){
580                                 TangoWebappHelpers.error(err);
581                             })
582 
583                     })(host, device);
584         },
585         /**
586          * Overrides attrs_monitor_view.addAttribute by adding state update
587          *
588          * @param {TangoAttribute} attr
589          */
590         addAttribute:function(attr){
591             if(webix.ui.attrs_monitor.prototype.addAttribute.apply(this, arguments)) {
592                 this.state.updateState(attr.id, false);
593             }
594         },
595             /**
596              *
597              * @param id
598              */
599         removeItem:function (id) {
600             webix.ui.attrs_monitor.prototype.removeItem.apply(this, arguments);
601             var state = this.state.getState();
602             delete state[id];
603             this.state.setState(state);
604         },
605             /**
606              *
607              * @param item
608              */
609         startPlot:function(item){
610             webix.ui.attrs_monitor.prototype.startPlot.apply(this, arguments);
611             this.state.updateState(item.id, true);
612         },
613             /**
614              *
615              * @param item
616              */
617         stopPlot:function(item){
618             webix.ui.attrs_monitor.prototype.stopPlot.apply(this, arguments);
619             this.state.updateState(item.id, false);
620         }
621     },TangoWebappPlatform.mixin.Stateful, attrs_monitor_view);
