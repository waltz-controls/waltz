  2 (function () {
  3     webix.protoUI({
  4         _change: {},
  5         _archive: {},
  6         _periodic: {},
  7         refresh: function () {
  8             var top = this.getTopParentView();
  9             var device = top._device;
 10 
 11             top.$$change.clearAll();
 12             top.$$archive.clearAll();
 13             top.$$periodic.clearAll();
 14             device.fetchAttrs()
 15                 .then(function (attrs) {
 16                     attrs.forEach(function (attr) {
 17                         var info = attr.info;
 18                         var id = attr.id;
 19                         this._change[id] = this.$$change.add({
 20                             name: info.name,
 21                             abs_change: info.events.ch_event.abs_change,
 22                             rel_change: info.events.ch_event.rel_change
 23                         });
 24 
 25                         this._archive[id] = this.$$archive.add({
 26                             masterId: id,
 27                             name: info.name,
 28                             abs_change: info.events.arch_event.abs_change,
 29                             rel_change: info.events.arch_event.rel_change,
 30                             period: info.events.arch_event.period
 31                         });
 32 
 33                         this._periodic[id] = this.$$periodic.add({
 34                             masterId: id,
 35                             name: info.name,
 36                             period: info.events.per_event.period
 37                         });
 38                     }.bind(top));
 39                 }).fail(TangoWebappHelpers.error);
 40         },
 41         apply: function () {
 42             var top = this.getTopParentView();
 43 
 44             var device = top._device;
 45 
 46             TangoWebappHelpers.iterate(device.attrs, function (attr, id) {
 47                 //change
 48                 var info = attr.info;
 49                 info.events.ch_event.abs_change = this.$$change.getItem(this._change[id]).abs_change;
 50                 info.events.ch_event.rel_change = this.$$change.getItem(this._change[id]).rel_change;
 51                 //archive
 52                 info.events.arch_event.abs_change = this.$$archive.getItem(this._archive[id]).abs_change;
 53                 info.events.arch_event.rel_change = this.$$archive.getItem(this._archive[id]).rel_change;
 54                 info.events.arch_event.period = this.$$archive.getItem(this._archive[id]).period;
 55                 //periodic
 56                 info.events.per_event.period = this.$$periodic.getItem(this._periodic[id]).period;
 57 
 58                 //TODO method in TangoAttribute
 59                 attr.set_attributes({
 60                     info: info
 61                 });
 62                 device.attrs.updateItem(id, info);
 63 
 64                 attr.putInfo().fail(TangoWebappHelpers.error);
 65             }.bind(top));
 66         },
