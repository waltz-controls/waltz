1 /**
  2  * MVC store adaptor for webix.DataCollection
  3  *
  4  * @type {WebixDataCollectionStorage}
  5  */
  6 var WebixDataCollectionStorage = MVC.Class.extend(
  7     /* @prototype */
  8     {
  9         _data: null,
 10         /**
 11          *
 12          * @param {Object} klass
 13          */
 14         init: function (klass) {
 15             this.storing_class = klass;
 16             this._data = (function () {
 17                 if (MVC.Browser.Rhino) //TODO :( get rid off Nashorn
 18                     return {};
 19                 else return new webix.DataCollection({
 20                     defaultData: klass.default_attributes
 21                 });
 22             })();
 23         },
 24         /**
 25          *
 26          * @param {Object} id
 27          */
 28         find_one: function (id) {
 29             return this._data.getItem(id);
 30         },
 31         /**
 32          *
 33          * @param {Object} obj
 34          */
 35         create: function (obj) {
 36             if(obj.Class.id !== 'id') obj.id = obj[obj.Class.id];
 37             obj[obj.Class.id] = this._data.add(obj);
 38         },
 39         /**
 40          * Updates stored instance
 41          *
 42          * @param id
 43          * @param attrs
 44          */
 45         update: function (id, attrs) {
 46             this._data.updateItem(id, attrs);
 47         },
 48         /**
 49          *
 50          * @param {Object} id
 51          */
 52         destroy: function (id) {
 53             this._data.remove(id);
 54         },
 55         /**
 56          * Finds instances using a test function.  If no test function is provided returns all instances.
 57          * @param {Function} f
 58          * @return {Array}
 59          */
 60         find: function (f) {
 61             f = f || function () {
 62                 return true;
 63             };
 64             if(typeof f === 'function')
 65                 return this._data.find(f);
 66             else
 67                 return this._data.getItem(f);//assume f is an id
 68         },
 69         /**
 70          * Clears instances
 71          */
 72         clear: function () {
 73             this._data.clearAll();
 74         },
 75         /**
 76          * Returns if there is no instances
 77          * @return {Boolean}
 78          */
 79         is_empty: function () {
 80             return this._data.count() === 0;
 81         }
 82     });
