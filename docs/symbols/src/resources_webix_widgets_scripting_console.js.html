 12         name: "codemirror_textarea",
 13         editor: null,
 14         getValue: function () {
 15             return this.editor.getValue();
 16         },
 17         setValue: function (value) {
 18             if (!value || typeof value !== 'string') return;
 19             this.editor.setValue(value);
 20         },
 21         $init: function () {
 22             this.$ready.push(function () {
 23                 this.attachEvent('onAfterRender', function () {
 24                     if(MVC.env().match(/development|test/)) console.time('CodeMirror render');
 25                     var value;
 26                     if(this.editor != null){
 27                         value = this.editor.getValue();
 28                     }
 29                     this.editor = CodeMirror.fromTextArea(this.getInputNode(),{
 30                         extraKeys: {"Ctrl-Space": "autocomplete"},
 31                         lineNumbers: true,
 32                         gutter: true,
 33                         lineWrapping: true
 34                     });
 35 
 36                     this.setValue(value);
 37                     if(MVC.env().match(/development|test/)) console.timeEnd('CodeMirror render');
 38                     // ~ 20ms
 39                 }.bind(this));
 40 
 41             }.bind(this));
 42         },
 43         defaults: {
 44             tooltip: 'Autocomplete: ctrl+space'
 45         }
 46     }, webix.ui.textarea);

180      * @type {webix.protoUI}
181      */
182     var scripting_console = webix.protoUI({
183         name: 'scripting_console',
184         /**
185          * @return {UserScript}
186          */
187         save:function(){
188             if(!this.isVisible() || this.$destructed) return;
189 
190             if(!this.$$('script_name').validate()) return null;
191             var name = this.$$('script_name').getValue().trim();
192             var code = this.$$('script_code').getValue();
193 
194             var script = UserScript.find_one(name);
195             var attrs = {
196                 name: name,
197                 code: code
198             };
199 
200             if (script == null)
201                 script = new UserScript(attrs);
202             else
203                 script.update_attributes(attrs);
204 
205             TangoWebappHelpers.logWithPopup("Script " + script.name + " is saved!" );
206 
207             return script;
208         },
209         /**
210          * @return {UserScript}
211          */
212         remove:function(){
213             if(!this.$$('script_name').validate()) return null;
214             var name = this.$$('script_name').getValue().trim();
215 
216             var script = UserScript.find_one(name);
217             script.destroy();
218             return script;
219         },
220         execute: function () {
221             if(!this.isVisible() || this.$destructed) return;
222 
223             var script = this.save();
224             if(script == null) return;
225             //TODO UserAction
226             var $$output = this.$$('output');
227             $$output.showProgress({
228                 type: "icon"
229             });
230 
231             UserAction.executeUserScript(script)
232                 .then(function (result) {
233                     //TODO OK NOK etc
234 
235                     $$output.setValue(result);
236                     $$output.hideProgress();
237                 })
238                 .fail(function (err) {
239                     //TODO color analyze etc
240                     $$output.setValue(err.errors);
241                     $$output.hideProgress();
242                 });
243         },
244         _ui: function () {
245             return {
246                 rows: [
247                     {
248                         gravity: 4,
249                         multi: true,
250                         cols: [
251                             {
252                                 header: '<span class="webix_icon fa-book"></span> Scripts',
253                                 body: scripts_list
254                             },
255                             {
256                                 gravity: 4,
257                                 body: {
258                                     rows: [
259                                         upper_toolbar,
260                                         script_code
261                                     ]
262                                 }
263                             }
264                         ]
265                     },
266                     {
267                         view: 'resizer'
268                     },
269                     lower_toolbar,
270                     output
271                 ]
272             }
273         },
274         $init: function (config) {
275             webix.extend(config, this._ui());
276 
277             this.$ready.push(function () {
278                 this.$$('scripts_list').data.sync(UserScript.store._data);
279                 this.$$('script_code').bind(UserScript.store._data);
280                 this.$$('script_name').bind(UserScript.store._data);
281 
282                 webix.extend(this.$$('output'), webix.ProgressBar);
283             }.bind(this));
284         }
285     }, webix.IdSpace, webix.ui.layout);

297     var stateful_scripting_console = webix.protoUI({
298         name:'stateful_scripting_console',
299         /**
300          *
301          * @param {WidgetState} state
302          */
303         restoreState:function(state){
304             var data = state.getState();
305             for(var script in data){
306                 UserScript.create_as_existing({
307                     name: script,
308                     code: data[script]
309                 })
310             }
311         },
312         /**
313          * Overrides scripting_console.save by adding state update
314          *
315          */
316         save:function(){
317             var script = webix.ui.scripting_console.prototype.save.apply(this, arguments);
318             if(script == null) return null;
319             var state = Object.create(null);
320             state[script.name] = script.code;
321             this.state.updateState(state);
322             return script;
323         },
324         /**
325          * Overrides scripting_console.delete by adding state update
326          *
327          */
328         remove:function(){
329             var script = webix.ui.scripting_console.prototype.remove.apply(this, arguments);
330             if(script == null) return null;
331             var data = this.state.getState();
332             delete data[script.name];
333             this.state.setState(data);
334             return script;
335         }
336     }, TangoWebappPlatform.mixin.Stateful, scripting_console);
