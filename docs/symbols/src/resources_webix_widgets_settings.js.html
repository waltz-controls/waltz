260                                     {
261                                         view: 'text',
262                                         id: "new-tango-host",
263                                         name: 'new_tango_host',
264                                         placeholder: 'localhost:10000',
265                                         invalidMessage: "Value does not match pattern: host:port",
266                                         validate: function (value) {
267                                             return /\w+:\d+/.test(value);
268                                         },
269                                         suggest: [] //TODO
270                                     },
271                                     {
272                                         view: 'button',
273                                         type: 'icon',
274                                         icon: 'plus-square-o',
275                                         width: 32,
276                                         click: function () {
277                                             var form = this.getFormView();
278                                             var isValid = form.validate();
279                                             if (!isValid) return;
280 
281                                             PlatformApi.UserContextController().add_tango_host(form.elements.new_tango_host.getValue());
282                                         }
283                                     }
284                                 ]
285                             }
286                         ]
287                     },
288                     {
289                         gravity: 6,
290                         id: 'tango_hosts',
291                         view: 'list',
292                         autoheight: true,
293                         select: true,
294                         template: "<span class='webix_icon fa-minus-square-o remove_tango_host'></span> #id#",
295                         on: {
296                             onItemClick: function (id) {
297                                 PlatformContext.tango_hosts.setCursor(id);
298                             }
299                         },
300                         onClick: {
301                             remove_tango_host: function (event, id) {
302                                 PlatformApi.UserContextController().delete_tango_host(id);
303                             }
304                         }
305                     }
306                 ]
307             };
308         },
309         $init: function (config) {
310             webix.extend(config, this._ui());
311 
312             this.$ready.push(function () {
313                 var data = [];
314                 var context = PlatformContext.UserContext;
315 
316                 for (var tango_host in context.tango_hosts) {
317                     if (!context.tango_hosts.hasOwnProperty(tango_host)) continue;
318 
319                     data.push({
320                         id: tango_host
321                     });
322                 }
323 
324                 $$('settings').$$('tango_hosts').parse(data);
325             });
326         },
327         defaults: {
328             minWidth: 240,
329             on: {
330                 "user_context_controller.found subscribe": function (event) {
331                     var data = [];
332                     var context = event.data;
333 
334                     for (var tango_host in context.tango_hosts) {
335                         if (!context.tango_hosts.hasOwnProperty(tango_host)) continue;
336 
337                         data.push({
338                             id: tango_host
339                         });
340                     }
341 
342                     $$('settings').$$('tango_hosts').parse(data);
343                 },
344                 "user_context_controller.add_tango_host subscribe": function (event) {
345                     $$('settings').$$('tango_hosts').add({
346                         id: event.data
347                     });
348                     var rest = PlatformContext.rest;
349                     rest.fetchHost(event.data)
350                         .then(function (tango_host) {
351                             return tango_host.fetchDatabase();
352                         }).then(function (db) {
353                         TangoWebappHelpers.log(db.device.host.id + " has been added.");
354                     }).fail(function (host) {
355                         TangoWebappHelpers.error("Failed to load Tango host[" + host.id + "]");//TODO errors
356                     });
357                 },
358                 "user_context_controller.delete_tango_host subscribe": function (event) {
359                     $$('settings').$$('tango_hosts').remove(event.data);
360 
361                     //TODO do we need to remove tango_host from context here?
362                 },
363                 "user_context_controller.destroy subscribe": function () {
364                     $$('settings').$$('tango_hosts').clearAll();
365                     $$('settings').$$('tango_hosts').refresh();
366                 }
367             }
368         }
369     }, TangoWebappPlatform.mixin.OpenAjaxListener, webix.ui.layout);
