1 /** @module DeviceMonitorView */
  2 (function () {
  3     /**
  4      * @type {webix.protoUI}
  5      */
  6     var device_monitor_header = webix.protoUI({
  7         _last_state: "UNKNOWN",
  8         name: "device_monitor_header",
  9         setValues: function (values, force) {
 10             webix.html.removeCss(this.getNode(), this._last_state, true);
 11             webix.html.addCss(this.getNode(), this._last_state = values.value, true);
 12             webix.ui.template.prototype.setValues.call(this, values, force);
 13         }
 14     }, webix.ui.template);
 15 
 16     /**
 17      * @type {webix.protoUI}
 18      */
 19     var device_monitor = webix.protoUI({
 20         _add_scalars: function(scalars){
 21             this.$$('attributes').parseScalars(scalars);
 22             this.$$('attributes').addDevice(this.config.device);
 23             //reshow progress as it is removed by parse
 24             // this.$$('scalar').showProgress({
 25             //     type: "icon"
 26             // });
 27         },
 28         /**
 29          * Loads device attributes
 30          *
 31          * @param {TangoDevice} device
 32          */
 33         loadAttributes: function (device) {
 34             console.time('loadAttributes')
 35             //TODO sync with device.attrs
 36             return device.fetchAttrs().then(function (attrs) {
 37                 var scalars = attrs.filter(function(attr){
 38                     return attr.info.data_format === 'SCALAR';
 39                 });
 40 
 41                 this._add_scalars(scalars);
 42 
 43                 var others = attrs.filter(function(attr){
 44                     return attr.info.data_format !== 'SCALAR';
 45                 });
 46 
 47                 others.forEach(function (attr) {
 48                     //TODO optimize
 49                     setTimeout(function () {
 50                         this.$$('attributes').addAttribute(attr);
 51                     }.bind(this), 10);
 52                 }.bind(this));
 53 
 54                 console.timeEnd('loadAttributes');
 55             }.bind(this))
 56                 .fail(function (err) {
 57                 TangoWebappHelpers.error(err);
 58             });
 59         },
 60         _ui_header: function (device) {
 61             return {
 62                 rows: [
 63                     {
 64                         id: "state",
 65                         view: "device_monitor_header",
 66                         template: function(obj){
 67                             return device.alias + "[" + device.name + "]: " + obj.value;
 68                         },
 69                         type: "header",
 70                         data: {
 71                             value: "UNKNOWN"
 72                         }
 73                     },
 74                     {
 75                         view: 'fieldset',
 76                         label: 'Status',
 77                         body: {
 78                             id: 'status',
 79                             template: "#value#",
 80                             data: {
 81                                 value: 'N/A'
 82                             }
 83                         }
 84                     }
 85                 ]
 86             };
 87         },
 88         _ui: function (device) {
 89             return {
 90                 rows: [
 91                     this._ui_header(device),
 92                     {view:'resizer'},
 93                     TangoWebapp.ui.newAttrsMonitorView({
 94                         id:'attributes',
 95                         gravity:4
 96                     })
 97                 ]
 98             };
 99         },
100         name: "device_monitor",
101         $init: function (config) {
102             webix.assert(config.device !== undefined, "DeviceMonitor requires device to be defined!");
103             webix.extend(config, this._ui(config.device));
104 
105             this.$ready.push(function () {
106                 this.$$('attributes').$$('scalars').hideColumn('device_id');
107                 this.$$('attributes').$$('scalars').hideColumn('remove');
108 
109                 this.$$('attributes').$$('scalars').data.attachEvent('onDataUpdate',function(id, update){
110                     if(this.config.device.attrs.getItem(id).name === 'State'){
111                         this.$$('state').setValues(update);
112                     }
113                     if(this.config.device.attrs.getItem(id).name === 'Status'){
114                         this.$$('status').setValues(update);
115                     }
116                 }.bind(this));
117 
118                 this.loadAttributes(config.device)
119                     .then(function () {
120                         this.$$('attributes').start();
121                         // return this.run();
122                     }.bind(this));
123             }.bind(this));
124 
125             // this.$ready.push(this.start.mvc_bind(this));
126         }
127     }, webix.IdSpace, webix.EventSystem,
128         TangoWebappPlatform.mixin.DeviceSetter, TangoWebappPlatform.mixin.TabActivator,
129         webix.ui.layout);

134             close: true,

137                 view: "device_monitor",

139                 device: config.device
140             }
141         }
142     };
143 })();
