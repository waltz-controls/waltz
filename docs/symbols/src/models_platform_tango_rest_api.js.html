1 /**
  2  * Model tango_rest_api
  3  *
  4  * @type {TangoRestApi}
  5  */
  6 TangoWebappPlatform.TangoRestApi = MVC.Model.extend('tango_rest_api',
  7     /** @Static */
  8     {
  9         _api_version: 'rc4',
 10         id: "url",
 11         attributes: {
 12             url: 'string'
 13         },
 14         default_attributes: {}
 15     },
 16     /** @Prototype */
 17     {
 18         promise: null,
 19         req_ids: null,
 20         /**
 21          *
 22          * @constructor
 23          * @param params
 24          */
 25         init: function (params) {
 26             if (!params.url) throw "Bad argument: url is expected here";
 27             this._super(params);
 28             this.req_ids = [];
 29             this.promise = webix.promise;
 30         },
 31         /**
 32          *
 33          * @returns {TangoRestApiRequest}
 34          */
 35         request: function () {
 36             var request = new TangoWebappPlatform.TangoRestApiRequest({url: this.url + "/tango/rest/" + this.Class._api_version});
 37             this.req_ids.push(request.id);//TODO or should we store actual ref here
 38             return request;
 39         },
 40         /**
 41          *
 42          * @param {string} host - host
 43          * @param {int} port - port
 44          *
 45          * @event {OpenAjax} tango_webapp.rest_api.fetch_host
 46          * @return {Promise}
 47          */
 48         fetchHost: function (host) {
 49             return this.request().hosts(host.replace(':', '/')).get()
 50                 .then(function (resp) {
 51                         var newHost = new TangoWebappPlatform.TangoHost(MVC.Object.extend(resp, {
 52                             id: host,
 53                             rest: this,
 54                             is_alive: true
 55                         }));
 56                         OpenAjax.hub.publish("tango_webapp.tango_host_loaded", {data: newHost});
 57                         return newHost;
 58                     }.bind(this)
 59                 ).fail(function (resp) {
 60                     var newHost = new TangoWebappPlatform.TangoHost({
 61                         id: host,
 62                         errors: resp.errors,
 63                         rest: this,
 64                         is_alive: false
 65                     });
 66                     OpenAjax.hub.publish("tango_webapp.tango_host_loaded", {data: newHost});
 67                     throw newHost;
 68                 }.bind(this));
 69         },
 70         /**
 71          *
 72          * @event {OpenAjax} tango_rest_api.is_alive
 73          * @event {OpenAjax} tango_rest_api.is_not_alive
 74          * @return {Promise}
 75          */
 76         isAlive: function () {
 77             var request = this.request();
 78             return request.get()
 79                 .then(function () {
 80                     this.publish("is_alive", {data: this});
 81                     return this;
 82                     }.bind(this)
 83                 ).fail(
 84                     function () {
 85                         this.publish("is_not_alive", {data: this});
 86                         throw this;
 87                     }.bind(this)
 88                 );
 89         }
 90     }
 91 );
 92 
 93 if (window['TangoRestApi'] === undefined)
 94     TangoRestApi = TangoWebappPlatform.TangoRestApi;
