1 /**
  2  *
  3  * @type {TangoHost}
  4  */
  5 TangoWebappPlatform.TangoHost = MVC.Model.extend("tango_host",
  6     /** @Static */
  7     {
  8         attributes: {
  9             host: "string",
 10             port: "number",
 11             name: "string",
 12             id: "string", //host:port
 13             info: "string[]",
 14             is_alive: 'boolean'
 15         },
 16         default_attributes: {
 17             id: 'not selected',
 18             host: 'unknown',
 19             port: 0,
 20             name: 'unknown',
 21             info: [],
 22             is_alive: false
 23         }
 24     },
 25     /** @Prototype */
 26     {
 27         rest: null,
 28         database: null,
 29         toString: function () {
 30             return this.id;
 31         },
 32         toUrl: function () {
 33             return this.host + "/" + this.port;
 34         },
 35         /**
 36          *
 37          * @event {OpenAjax} tango_webapp.device_loaded
 38          *
 39          * @param name
 40          * @return {Promise} device
 41          */
 42         fetchDevice: function (name) {
 43             return this.fetchDatabase()
 44                 .then(function (db) {
 45                     return webix.promise.all(
 46                         [
 47                             db.getDeviceInfo(name),
 48                             db.getDeviceAlias(name).fail(function(){
 49                                 return "";
 50                             })
 51                         ]);
 52                 })
 53                 .then(function (info) {
 54                     var device = new TangoWebappPlatform.TangoDevice({
 55                         info: info[0],
 56                         alias: info[1],
 57                         id: this.id + "/" + name,
 58                         name: name,
 59                         host: this
 60                     });
 61                     OpenAjax.hub.publish("tango_webapp.device_loaded", {data: device});
 62                     return device;
 63                 }.bind(this));
 64         },
 65         /**
 66          *
 67          * @event {OpenAjax} tango_webapp.database_loaded
 68          * @return {Promise} database
 69          */
 70         fetchDatabase: function () {
 71             return this.rest.request().hosts(this.toUrl()).devices(this.name).get()
 72                 .then(function (resp) {
 73                         //jmvc fails to set "attributes" due to already existing function in the model
 74                         delete resp.attributes;
 75 
 76                         var device = new TangoWebappPlatform.TangoDevice(MVC.Object.extend(resp, {
 77                             id: this.id + "/" + this.name,
 78                             name: this.name,
 79                             host: this
 80                         }));
 81 
 82                     this.is_alive = true;
 83                     this.errors = [];
 84                     OpenAjax.hub.publish("tango_webapp.tango_host_loaded", {data: this});
 85                         return device;
 86                     }.bind(this)
 87                 ).fail(function (resp) {
 88                     this.is_alive = false;
 89                         this.add_errors(resp.errors);
 90                         throw resp;
 91                     }.bind(this)
 92                 ).then(function (device) {
 93                     OpenAjax.hub.publish("tango_webapp.device_loaded", {data: device});//TODO use PlatformContext directly?
 94                     this.database = new TangoWebappPlatform.TangoDatabase({
 95                         id: device.id,
 96                         device: device,
 97                         info: this.info
 98                     });
 99                     OpenAjax.hub.publish("tango_webapp.database_loaded", {data: this.database});
100                     return this.database;
101                 }.bind(this));
102         }
103     }
104 );
